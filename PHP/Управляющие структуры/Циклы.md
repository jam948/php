[[Управляющие структуры]]
### while 
while (условие равно true) {
   код будет выполняться;
}

### do while
do {
  выполняется код;
} while (условие равно true); 

### for 
for (expr1; expr2; expr3) {
   выполняемый код (тело цикла);
}


### foreach
Оператор **foreach** дает возможность перебирать элементы массивов.

foreach (array as $value) {
    код для выполнения;
}
//или
foreach (array as $key => $value) {
    код для выполнения;
}

$cities = array("Лондон", "Париж", "Мадрид");
foreach ($cities as $city) {
    echo $city."\n";
}

/* выведет:
Лондон
Париж
Мадрид
*/

### break
< ? php
$arr = array(2, 4, 6, 7, 8, 10, 12);
foreach ($arr as $val) {
    if ($val % 2 != 0) {
        break;
    } echo $val."\n";
}

/* выведет:
2
4
6
*/
?>

### continue
< ?php
$arr = array(6, 7, 8, 9, 10, 11, 12);
foreach ($arr as $val) {
    if ($val % 2 != 0) {
        continue;   
    } echo $val."\n";
}

/* выведет:
6
8
10
12
*/
